// cloud_server/ClientManager.js
const WebSocket = require('ws');
const crypto = require('crypto');

class ClientManager {
    constructor() {
        this.sessions = new Map();
        this.clients = new Map();
        this.sessionStats = new Map();
        
        this.settings = {
            maxClientsPerSession: 10,
            sessionTimeout: 3600000,
            clientHeartbeatInterval: 30000
        };
    }

    createSession(sessionId, hostInfo) {
        const session = {
            id: sessionId,
            host: hostInfo,
            clients: new Map(),
            created: Date.now(),
            settings: {
                maxPlayers: 10,
                radarMode: '2d',
                dataSharing: 'full'
            }
        };
        
        this.sessions.set(sessionId, session);
        return session;
    }

    addClient(sessionId, clientWs, clientInfo) {
        const session = this.sessions.get(sessionId);
        if (!session) throw new Error('Session not found');

        const clientId = crypto.randomBytes(8).toString('hex');
        const client = {
            id: clientId,
            ws: clientWs,
            info: clientInfo
        };

        session.clients.set(clientId, client);
        this.clients.set(clientId, { sessionId, client });
        
        this.notifyHost(sessionId, 'client_joined', {
            clientId,
            clientInfo,
            clientCount: session.clients.size
        });
        
        return client;
    }

    notifyHost(sessionId, type, data) {
        const session = this.sessions.get(sessionId);
        if (!session?.host?.ws) return;

        if (session.host.ws.readyState === WebSocket.OPEN) {
            session.host.ws.send(JSON.stringify({
                type: `host_${type}`,
                ...data
            }));
        }
    }
}
