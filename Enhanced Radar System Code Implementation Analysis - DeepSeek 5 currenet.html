        function drawRadar() {
            // Clear canvas
            ctx.clearRect(0, 0, canvas.width, canvas.height);

            // Draw players
            players.forEach(player => {
                // Convert game coordinates to radar coordinates (simplified)
                const radarX = (player.Position.X + 1000) * (canvas.width / 2000);
                const radarY = (player.Position.Y + 1000) * (canvas.height / 2000);

                // Draw the player
                ctx.beginPath();
                ctx.arc(radarX, radarY, 5, 0, 2 * Math.PI);
                ctx.fillStyle = player.Team === 2 ? 'blue' : 'red';
                ctx.fill();
            });
        }

        // Initial draw
        drawRadar();
